25  package ru.ewc.decita.input;
26  
27  import java.util.ArrayList;
28  import java.util.List;
29  import ru.ewc.decita.Coordinate;
30  import ru.ewc.decita.DecisionTable;
31  import ru.ewc.decita.EqualsCondition;
32  import ru.ewc.decita.Rule;
33  
34  /**
35   * I am the unified source for building {@link DecisionTable}s. My main responsibility is to store
36   * all the data needed to construct {@link Rule}s and fill the {@link DecisionTable}.
37   *
38   * @since 0.2
39   */
40  public final class RawContent {
41      /**
42       * The name of the source table.
43       */
44      private final String table;
45  
46      /**
47       * Array of String values that form the Conditions part of a {@link DecisionTable}.
48       */
49      private final String[][] conditions;
50  
51      /**
52       * Array of String values that form the Outcomes/Actions part of a {@link DecisionTable}.
53       */
54      private final String[][] outcomes;
55  
56      /**
57       * Ctor.
58       *
59       * @param conditions A 2D-array of Strings describing the Conditions part of the
60       *  {@link DecisionTable}.
61       * @param outcomes A 2D-array of Strings describing the Outcomes part of the
62       *  {@link DecisionTable}.
63       * @param table Name of the source table.
64       */
65      public RawContent(final String[][] conditions, final String[][] outcomes, final String table) {
66          this.table = table;
67          this.conditions = conditions.clone();
68          this.outcomes = outcomes.clone();
69      }
70  
71      /**
72       * Answers the table name.
73       *
74       * @return The table name.
75       */
76      public String tableName() {
77          return this.table;
78      }
79  
80      /**
81       * Creates a {@link DecisionTable} from the raw contents.
82       *
83       * @return The {@link DecisionTable} created from the source file.
84       */
85      public DecisionTable asDecisionTable() {
86          final List<Rule> rules = new ArrayList<>(this.conditions[0].length - 1);
87          for (int column = 1; column < this.conditions[0].length; column += 1) {
88              final Rule rule = new Rule();
89              for (final String[] condition : this.conditions) {
90                  rule.withCondition(
91                      new EqualsCondition(
92                          coordinateFrom(condition[0]),
93                          coordinateFrom(condition[column])
94                      )
95                  );
96              }
97              for (final String[] outcome : this.outcomes) {
98                  rule.withOutcome(
99                      outcome[0],
100                     outcome[column]
101                 );
102             }
103             rules.add(rule);
104         }
105         return new DecisionTable(rules);
106     }
107 
108     /**
109      * Creates a {@link Coordinate} based on a string representation.
110      *
111      * @param coordinate String representation of a coordinate.
112      * @return A concrete {@link Coordinate} based on given string representation.
113      */
114     private static Coordinate coordinateFrom(final String coordinate) {
115         final String[] split = coordinate.split("::");
116         return new Coordinate(split[0], split[1]);
117     }
118 }
