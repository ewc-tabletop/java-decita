25  package ru.ewc.decita;
26  
27  import java.util.Map;
28  import java.util.stream.StreamSupport;
29  
30  /**
31   * I am a collection of {@link Rule}s used to compute any kind of decision. My main responsibility
32   * is to orchestrate the computation and choose the only satisfied {@link Rule}.
33   *
34   * @since 0.1
35   */
36  public final class DecisionTable implements Locator {
37      /**
38       * A collection of table's {@link Rule}s.
39       */
40      private final Iterable<Rule> rules;
41  
42      /**
43       * The special {@link Rule} that gets satisfied only if no other {@link Rule} is satisfied.
44       * Basically it means that for every table there's always one satisfied {@link Rule}.
45       */
46      private final Rule elserule;
47  
48      /**
49       * Ctor.
50       *
51       * @param rules A collection of {@link Rule}s for this table.
52       */
53      public DecisionTable(final Iterable<Rule> rules) {
54          this.rules = rules;
55          this.elserule = new Rule().withOutcome("outcome", "undefined");
56      }
57  
58      @Override
59      public String fragmentBy(final String fragment, final ComputationContext context)
60          throws DecitaException {
61          return this.outcome(context).getOrDefault(fragment, "undefined");
62      }
63  
64      @Override
65      public Map<String, String> outcome(final ComputationContext context) throws DecitaException {
66          for (final Rule rule : this.rules) {
67              rule.check(context);
68          }
69          return StreamSupport.stream(this.rules.spliterator(), false)
70              .filter(Rule::isSatisfied)
71              .findFirst().orElse(this.elserule)
72              .outcome();
73      }
74  }
